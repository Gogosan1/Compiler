Грамматика

G[<ЦК> = <целочисленная константа>]:
VT = { ‘const ’, ‘double ’, ‘a’…’z’, ‘A’…’Z’, ‘0’…’9’, ‘.’, ‘=’, ‘+’, ‘-‘, ‘;’, ‘_’ }
VN = { <CONST>, , DOUBLE, NAME, NAMEREM, NUMBER, INT, INTREM, DECIMAL, DECIMALREM, END, letter, digit }
P = {
 1. <CONST> → ‘const’ DOUBLE
 2. DOUBLE → 'double ' NAME
 3. NAME → letter NAMEREM
 4. NAMEREM → letter NAMEREM | '=' NUMBER
 5. NUMBER → ['+'|'-'] INT
 7. INT → digit INTREM
 8. INTREM → digit INTREM | '.' DECIMAL | ';'
 9. DECIMAL → digit DECIMALREM
 10. DECIMALREM → digit DECIMALREM | ';'
 11. letter → ‘a’ | ‘b’ | … | ‘z’ | ‘A’ | ‘B’ | … | ‘Z’
 12. digit → ‘0’ | ‘1’ | … | ‘9’
}
